var pagecookie ={
        setCookie:function(name,value,time){
            var strsec = this.getsec(time);
            var exp = new Date();
            exp.setTime(exp.getTime() + strsec*1);
            document.cookie = name + "=" + escape (value) + ";expires=" + exp.toGMTString();
        },
        getCookie:function(name){
            var arr,reg=new RegExp("(^| )" +name+"=([^;]*)(;|$)");
            if(arr=document.cookie.match(reg)) return unescape(arr[2]);
                else return null;
        },
        delCookie:function(name){
            var exp = new Date();
            exp.setTime(exp.getTime() - 1);
            var cval=this.getCookie(name);
            if(cval!=null) document.cookie= name + "="+cval+";expires="+exp.toGMTString();
        },
        getsec:function(str){
            var str1=str.substring(1,str.length)*1; 
            var str2=str.substring(0,1); 
            if (str2=="s"){
                return str1*1000;
            }else if (str2=="m"){
                return str1*60*1000;
            }else if (str2=="h"){
                return str1*60*60*1000;
            }else if (str2=="d"){
                return str1*24*60*60*1000;
            }
        }
    }

function changebatchsearchtip(){
    var batchtipname = pagecookie.getCookie("batchtipname");
    var batchtipmask = pagecookie.getCookie("batchtipmask");

    if(batchtipmask != "ques"){
        $(".ques_mark").hide(); 
        $(".tips_content").show();        
    }else if(!batchtipname && !batchtipmask){
        $(".ques_mark").hide(); 
        $(".tips_content").show();
    }else if(batchtipmask == "ques" || batchtipmask == "null" ){ 
        $(".ques_mark").show(); 
        $(".tips_content").hide();
    }else{
        var thiso = $("#"+batchtipname);
        $(".tips_infor").addClass("hidden");
        thiso.removeClass("hidden");
        
        var tip_num = batchtipname.replace("tip","");
        if(tip_num < 0) tip_num = 0;
        if(tip_num > 0){
            $("#use_xts").hide();
            $("#use_help").show();
            $("#tip_num").html(tip_num);
        }else{
            $("#use_xts").show();
            $("#use_help").hide();
        }
        $("#tip"+tip_num).removeClass("hidden");
        togglebatchtipnum(tip_num);
    }
}
function togglebatchtipnum(tip_num){
        if(tip_num < 0) tip_num = 0;
        if (tip_num >= 0){$(".back").show();}
        if (tip_num == 0){$(".back").hide();}
        if (tip_num > 0 ){$(".next").show();}
        if (tip_num >= 3){$(".next").hide();}
        if (tip_num == 0){$(".help").show();}
        if (tip_num != 0){$(".help").hide();}
}

        if (3 === time_arr.length) {
            var year = parseInt(time_arr[0]) - 0;
            var month = parseInt(time_arr[1]) - 0 - 1;//月份是从0开始的
            var day = parseInt(time_arr[2]) - 0;
            if(fix_time) {//有时间段
                var fix_arr = fix_time.split(':');
                if (3 === fix_arr.length) {
                    var hour = parseInt(fix_arr[0]) - 0;
                    var minute = parseInt(fix_arr[1]) - 0;
                    var second = parseInt(fix_arr[2]) - 0;
                    time_stamp = (new Date(year, month, day, hour, minute, second)).getTime()/1000;
                }
            }else{
                time_stamp = (new Date(year, month, day)).getTime()/1000;
            }
         }
     }else{
         time_stamp = "";
     }
   return time_stamp;
}

        if(len != len1){
            text_obj.val(trim_val);
        }
        error_obj.html("");
        if(len > 0){
            if(len<2){
                  falg_array.push(0);
                  error_obj.html("搜索词太短，请重新输入");
            }else if(len>70){
                falg_array.push(0);
                error_obj.html("搜索词太长，请重新输入");
            }else{
                falg_array.push(1);
            }
        }else{
            null_count +=1;
        }
        if($.trim(error_obj.html())!=""){
            error_obj.removeClass("hidden");
        }else{
            error_obj.addClass("hidden");
        }
    }
    for (var i = 0; i < falg_array.length; i++) {
            if (falg_array[i] == 0) {
                flag = 2;break;
            }
    }
    if(null_count == count){
        flag = 3;
    }
    return flag;
}

function filterPrice(){

    if(re.test(price)){
       $(this).val(''); return;
    }
    price = $(this).val();
    if(price.length>15){
        var new_value = price.substr(0,15);
        $(this).attr('value',new_value);
    };

    price = $(this).val();
    if(re.test(price)){
        var new_value = price.substr(0,price.search(re));
        $(this).attr('value',new_value);
    }
    re = /\.[0-9]{3}/i;
    price = $(this).val();
    if(re.test(price)){
        var new_value = price.substr(0,price.indexOf('.')+3);
        $(this).attr('value',new_value);return;
    }
    re = /\.[0-9]*\.+/i;
    price = $(this).val();
    if(re.test(price)){
        var new_value = price.substr(0,price.indexOf('.')+1);
        if(!isNaN(price.substr(price.indexOf('.')+1,1))){
            new_value = price.substr(0,price.indexOf('.')+2);
            if(!isNaN(price.substr(price.indexOf('.')+2,1))){
                new_value = price.substr(0,price.indexOf('.')+3);
            }
         }
        $(this).attr('value',new_value);return;
    }
    return;
}
